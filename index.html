JailSystems.lua
-- correctional officer role, part of the jail's
-- job system. real snippet-style example
local Jobs = require(script.Parent.JobConfig)
local DataStore = game:GetService("DataStoreService")

local function assignJob(player, jobName)
  local job = Jobs[jobName]
  if not job then return end

  player:SetAttribute("Job", jobName)
  giveTools(player, job.tools)
  print(player.Name .. " is now a " .. jobName)
end
